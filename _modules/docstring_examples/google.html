# -*- coding: utf-8 -*-
"""Example Google style docstrings.

This module demonstrates documentation as specified by the
`Google Python Style Guide`_.
Docstrings may extend over multiple lines. Sections are created
with a section header and a colon followed by a block of indented text.

Example:
    There are multiple ways to use `Example` or `Examples`, in this case we use
    a single colon `:` allowing us to use text.

    If we wish to display a codeblock you may use double-colon `::`
    ::

        Just like this

Example::

    my_string = "Alternatively you can just use a codeblock with double-colon"
    string2 = "This is just an example that will remain unexecuted"

Examples::

    >>> foo = "You can add chevrons to enable doctest on your codeblock"
    >>> " ".join(foo.split(" ")[::-1])
    codeblock your on doctest enable to chevrons addcan You

Section breaks are created by resuming unindented text. Section breaks
are also implicitly created anytime a new section starts.

.. _Google Python Style Guide: http://google.github.io/styleguide/pyguide.html
"""

    If the `__init__` method contains any notable behaviour it should be

    Raise:
        ValueError: raise on instantiation if input out of bound

    CLASS_LEVEL_VARS: str = "str"
    """The class attribute should be documented here."""

    def __init__(self, param1: int, param2: int, param3: str):
        if param1 > param2:
            raise ValueError()

    @property
    def readonly_property(self) -> str:
        """Properties should be documented in their getter method."""
        return "readonly_property"

    @property
    def readwrite_property(self) -> typing.List[str]:
        """Properties with both a getter and setter
        should only be documented in their getter method.

        If the setter method contains notable behavior, it should be
        documented here.
        """
        return ["readwrite_property"]

    @readwrite_property.setter
    def readwrite_property(self, value: str):
        self._value = value
